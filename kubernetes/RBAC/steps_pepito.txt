## RBAC
minikube start --vm-driver=none --extra-config=apiserver.authorization-mode=RBAC

## Create keys and sign
openssl genrsa -out pepito.key 2048
openssl req -new -key pepito.key -out pepito.csr -subj "/CN=pepito/O=dev"

kubectl config view
    certificate-authority: /home/fpepito/.minikube/ca.crt

sudo openssl x509 -req -in pepito.csr -CA /home/fjmoreno/.minikube/ca.crt -CAkey /home/fjmoreno/.minikube/ca.key -CAcreateserial -out pepito.crt -days 500
openssl x509 -in pepito.crt -noout -text

kubectl config current-context
minikube --> este user es admin

## Configure kubectl for user
kubectl cluster-info --> para ver el server
kubectl config set-cluster minikube --server=https://127.0.0.1:32776 --certificate-authority=/home/fjmoreno/.minikube/ca.crt
kubectl config set-credentials pepito --client-certificate=pepito.crt --client-key=pepito.key
kubectl config set-context pepito --cluster=minikube --user=pepito
kubectl config use-context pepito
