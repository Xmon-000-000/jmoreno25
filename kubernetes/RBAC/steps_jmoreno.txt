## RBAC
minikube start --vm-driver=none --extra-config=apiserver.authorization-mode=RBAC

## Create keys and sign
openssl genrsa -out jmoreno.key 2048
openssl req -new -key jmoreno.key -out jmoreno.csr -subj "/CN=jmoreno/O=dev"

kubectl config view
    certificate-authority: /home/fjmoreno/.minikube/ca.crt

sudo openssl x509 -req -in jmoreno.csr -CA /home/fjmoreno/.minikube/ca.crt -CAkey /home/fjmoreno/.minikube/ca.key -CAcreateserial -out jmoreno.crt -days 500
openssl x509 -in jmoreno.crt -noout -text

kubectl config current-context
minikube --> este user es admin

kubectl config set-cluster minikube --server=https://127.0.0.1:32776 --certificate-authority=/home/fjmoreno/.minikube/ca.crt
