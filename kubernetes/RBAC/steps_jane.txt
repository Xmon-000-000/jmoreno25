## RBAC
minikube start --vm-driver=none --extra-config=apiserver.authorization-mode=RBAC

## Create keys and sign
openssl genrsa -out jane.key 2048
openssl req -new -key jane.key -out jane.csr -subj "/CN=jane/O=dev"

kubectl config view
    certificate-authority: /home/fjane/.minikube/ca.crt

sudo openssl x509 -req -in jane.csr -CA /home/fjmoreno/.minikube/ca.crt -CAkey /home/fjmoreno/.minikube/ca.key -CAcreateserial -out jane.crt -days 500
openssl x509 -in jane.crt -noout -text

kubectl config current-context
minikube --> este user es admin
