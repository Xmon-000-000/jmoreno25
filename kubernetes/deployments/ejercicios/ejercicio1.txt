---
apiVersion: v1
kind: Namespace
metadata:
  name: deploy-lab
  labels:
    name: deploy-lab
---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: frontend
  namespace: deploy-lab
  labels:
    app: frontend
spec:
  replicas: 3
  selector:
    matchLabels:
      app: frontend
  template:
    metadata:
      labels:
        app: frontend
    spec:
      containers:
      - name: nginx
        image: nginx:no-existe
        ports:
        - containerPort: 80
  strategy:
    type: RollingUpdate
    rollingUpdate:
        maxUnavailable: 1
        maxSurge: 1
---
apiVersion: v1
kind: Service
metadata:
  name: frontend-svc
  namespace: deploy-lab
spec:
  selector:
    app: frontend
  ports:
    - protocol: TCP
      port: 80
      targetPort: 80
